    motion-objection

    Wrapping the dependency injection library Objection in RubyMotion.

    Install

    gem install motion-objection

    Basic Example

    class Car
  include Objection::Compose
  compose_with :engine, :brakes, factory: JSObjectFactory
end

class Engine
  include Objection::Compose
  singleton
  compose_with 'engine/crank_shaft', 'engine/rod'

  def shift(wat)
    puts "SHIFT!"
  end

  class CrankShaft
  end 

  class Rod
  end
end

class Brakes
  def brake!
  end
end

    Initializers

    class ViewController < NSObject
  include Objection::Compose
  initializer "initWithNibName:bundle:", "Home"

  attr_reader :name

  def initWithNibName(name, bundle: bundle)
    self.init
    self.tap do
      @name = name
    end
  end
end

    Injector

    Objection.default_injector = Objection.injector AppModule.new, SecondaryModule.new
